Makeover Monday: July 30, 2018

Comments

I created a slopegraph this week where I ordered the lines by their slope (largest negative slope at bottom and highest positive slope on top). I wanted to highlight how much countries changed in valuation, but I am not satisfied with the end result. It’s definitely a misleading graph because the data points are not ordered by their values. I wanted to try this out, and it took a lot of time to prepare the data so I didn’t have time to make something else. Also, for some reason, the lines are pixelated which make the plot even more unattractive. Oh, well. Live and learn.

Code 

# libraries ----
library(tidyr)
library(dplyr)
library(ggplot2)
library(data.world)



# data import ----

# datasets are referenced by their URL or path
dataset_key <- "https://data.world/makeovermonday/2018w31-big-mac-index"
# List tables available for SQL queries
tables_qry <- data.world::qry_sql("SELECT * FROM Tables")
tables_df <- data.world::query(tables_qry, dataset = dataset_key)

if (length(tables_df$tableName) > 0) {
  data_qry <- data.world::qry_sql(sprintf("SELECT * FROM `%s`", tables_df$tableName[[1]]))
  prelim_data <- data.world::query(data_qry, dataset = dataset_key)
}



# data prep ----

# 2018 data
dat <- prelim_data %>%
  filter(date == as.Date("2018-07-01")) %>%
  mutate(year = 2018) %>%
  filter(!is.na(usd_adjusted))

# 2017 data
dat2 <- prelim_data %>%
  filter(date == as.Date("2017-07-01")) %>%
  mutate(year = 2017) %>%
  filter(!is.na(usd_adjusted))

dat_comb<- rbind(dat, dat2)

dat_long <- dat_comb %>%
  select(year, usd_adjusted, currency_code)

dat_wide <- spread(dat_long, year, usd_adjusted) %>%
  mutate(slope = `2018` - `2017`) %>%
  arrange(slope) 

dat_wide$new_y17 <- numeric(37)
dat_wide$new_y18 <- numeric(37)
for(i in 1:(length(dat_wide$slope)-1)){
  dat_wide$new_y17[i+1] <- 0.207 + i*0.1
  dat_wide$new_y18[i+1] <- dat_wide$new_y17[i+1] + dat_wide$slope[i+1]
}
dat_wide$new_y17[1] <- dat_wide$`2017`[1]
dat_wide$new_y18[1] <- dat_wide$`2018`[1]

dat_long <- dat_wide %>%
  select(currency_code, `2017`, `2018`, new_y17, new_y18) %>%
  gather(year, value, new_y17:new_y18) %>%
  mutate(Year = as.numeric(substr(year, 6,7)))

dat_long$valued <- character(nrow(dat_long))
for(j in 1:nrow(dat_long)){
  if(dat_long$Year[j] == 17){
    if(dat_long$`2017`[j] > 0){
      dat_long$valued[j] <- "Overvalued"
    } else {
      dat_long$valued[j] <- "Undervalued"
    }
  } else if(dat_long$Year[j] == 18){
    if(dat_long$`2018`[j] > 0){
      dat_long$valued[j] <- "Overvalued"
    } else {
      dat_long$valued[j] <- "Undervalued"
    }
  }
}

dat_ann_17 <- dat_long %>%
  filter(Year == 17)

dat_ann_17$round17 <- as.character(round(dat_ann_17$`2017`, 3))

dat_ann_18 <- dat_long %>%
  filter(Year == 18)

dat_ann_18$round18 <- as.character(round(dat_ann_18$`2018`, 3))



# plot ----

ggplot(dat_long, aes(Year, value, group = currency_code)) +
  geom_line(size = 1, alpha = 1) +
  scale_x_continuous(limits = c(15, 20), breaks = 17:18, labels = c("2017", "2018")) +
  scale_color_manual(values = c("red", "blue")) +
  geom_text(data = dat_ann_17, aes(Year-0.15, value, label=round17), hjust = 1) +
  geom_text(data = dat_ann_17, aes(Year-1.25, value, label=currency_code), hjust = 0) +
  guides(colour = guide_legend(nrow = 1)) +
  geom_text(data = dat_ann_18, aes(Year+0.65, value, label=round18), hjust = 1) +
  geom_text(data = dat_ann_18, aes(Year+0.85, value, label=currency_code), hjust = 0) +
  ggtitle("Currency under/over valuation against the dollar",
          "Ordered by changes in Big Mac Index from July 2017 to July 2018") +
  labs(caption = "Source: The Economist") +
  theme_minimal() +
  theme(axis.text.y = element_blank(),
        axis.title = element_blank(),
        panel.grid = element_blank(), 
        legend.title = element_blank(),
        legend.position = c(0.17, 0.98),
        plot.caption = element_text(hjust = 0, size = 11))
