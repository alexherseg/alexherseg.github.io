Comments

I used a heatmap for this weeks Makeover Monday. I really like how it turned out; I think the heatmap does a good job of showing the income differences between the states.

Code 

# libraries ----
library(data.world)
library(dplyr)
library(ggplot2)
library(viridis)



# data import ----

# Datasets are referenced by their URL or path
dataset_key <- "https://data.world/makeovermonday/2018-w-3-u-s-household-income-distribution-by-state"
# List tables available for SQL queries
tables_qry <- data.world::qry_sql("SELECT * FROM Tables")
tables_df <- data.world::query(tables_qry, dataset = dataset_key)

if (length(tables_df$tableName) > 0) {
  data_qry <- data.world::qry_sql(sprintf("SELECT * FROM `%s`", tables_df$tableName[[1]]))
  prelim_data <- data.world::query(data_qry, dataset = dataset_key)
}



# data preparation ----

# order the income levels
prelim_data$income_level <- factor(prelim_data$income_level, 
                                   levels = c("Less than $10,000",
                                              "$10,000 to $14,999",
                                              "$15,000 to $19,999",
                                              "$20,000 to $24,999",
                                              "$25,000 to $29,999",
                                              "$30,000 to $34,999",
                                              "$35,000 to $39,999",
                                              "$40,000 to $44,999",
                                              "$45,000 to $49,999",
                                              "$50,000 to $59,999",
                                              "$60,000 to $74,999",
                                              "$75,000 to $99,999",
                                              "$100,000 to $124,999",
                                              "$125,000 to $149,999",
                                              "$150,000 to $199,999",
                                              "$200,000 or more"))

# get median income levels
dat <- prelim_data %>%
  arrange(year, state, income_level) %>%
  group_by(year, state) %>%
  mutate(cum_inc_lev = cumsum(percent_of_total),
         region = tolower(state)) %>%
  filter(cum_inc_lev > 0.5) %>%
  slice(n = 1)

# order states by income  
state_order <- dat %>%
  select(state, year, income_level) %>%
  tidyr::spread(key = year, value = income_level) %>%
  arrange(`2016`, `2015`, `2014`, `2013`, `2012`, `2011`, `2010`, `2009`)

dat$state <- factor(dat$state, levels = state_order$state)
  


# plot ----
p <- ggplot(dat, aes(year, state)) +
  geom_tile(aes(fill = income_level), color = "black") +
  scale_fill_viridis(name = "Median Income\nBetween:", 
                     option = "C", discrete = T, direction = -1) +
  scale_x_continuous(breaks = 2009:2016) +
  guides(fill = guide_legend(reverse=T)) +
  ggtitle("U.S. Median Household Income By State") +
  labs(caption = "Source: U.S. Census Bureau") +
  theme_minimal() +
  theme(axis.title = element_blank(), 
        plot.caption = element_text(hjust = 0))

# png
ggsave("C:/Users/alexherseg/Documents/blog/alexherseg.github.io/makeover_monday/03/15JAN2018.png",
       p, width = 6.5, height = 8)
